    def update_table(self, data=None):
        # Clear the existing rows in the table
        for row in self.tree.get_children():
            self.tree.delete(row)

        # Use the filtered data if provided, otherwise use the full inventory DataFrame
        data_to_display = data.copy() if data is not None else self.inventory_df.copy()

        # Add a row number to the DataFrame before displaying it in the Treeview
        data_to_display["Row Number"] = range(1, len(data_to_display) + 1)

        # Display the data in the Treeview with alternating tag configurations
        for index, row_data in data_to_display.iterrows():
            if index % 2 == 0:
                tag = "even_row"
            else:
                tag = "odd_row"

            self.tree.insert("", "end", values=row_data.tolist(), tags=(tag,))

        # Update row number label
        row_count = len(data_to_display)
        self.row_number_label.config(text=f"Total Rows: {row_count}")

        # Refresh the treeview to make sure it displays the updated data
        self.tree.update_idletasks()

    def save_to_excel(self):
        # Replace all NaN values with empty spaces
        self.inventory_df = self.inventory_df.replace(np.nan, "", regex=True)

        # Save the current inventory data to the Excel file, replacing NaN values with empty strings
        with pd.ExcelWriter("inventory_data.xlsx") as writer:
            self.inventory_df.fillna("", inplace=True)
            self.inventory_df.to_excel(writer, sheet_name="Inventory", index=False)

    def on_closing(self):
        # Save the data to the Excel file before exiting
        self.save_to_excel()
        self.destroy()

    def read_column(self, file_path, sheet_name, column_name):
        """
        Read a specific column from an Excel file and return its values as a list.
        Parameters:
            file_path (str): Path to the Excel file.
            sheet_name (str): Name of the sheet in the Excel file.
            column_name (str): Name of the column to read from the Excel sheet.
        Returns:
            list: A list containing the values from the specified column.
        """
        try:
            df = pd.read_excel(file_path, sheet_name=sheet_name)
            if column_name not in df.columns:
                raise ValueError(f"The column '{column_name}' does not exist in the sheet '{sheet_name}'.")
            return df[column_name].tolist()
        except Exception as e:
            print(f"An error occurred while reading the column: {str(e)}")
            return []

    def edit_product(self):
        # Get the selected item from the treeview
        selected_item = self.tree.selection()

        if not selected_item:
            # If no item has been selected, show a message box and return without editing
            messagebox.showinfo("No Item Selected", "Please select an item to edit.")
            return

        try:
            # Get the index of the selected item
            index = self.tree.index(selected_item)

            # Retrieve the previous data for the selected item from the DataFrame
            previous_data = self.inventory_df.iloc[index].to_dict()

            # Get the updated data from the entry fields
            edited_data = {
                "Main Category of Asset": self.maincategory_combobox.get(),
                "Sub Category of Asset": self.subcategory_combobox.get(),
                "Product ID": self.id_entry.get(),
                "Asset Name": self.assetname_entry.get(),
                "Supplier": self.supplier_entry.get(),
                "Location": self.location_combobox.get(),
                "Status": self.status_combobox.get(),
                "Asset Description": self.assetdescription_entry.get(),
                "Asset Type": self.asset_type_combobox.get(),
                "Brand": self.brand_entry.get(),
                "Model": self.model_entry.get(),
                "Serial Number": self.serial_entry.get(),
                "Quantity": self.quantity_entry.get(),
                "Purchase Date": self.purchase_date_entry.get(),
                "Title of Asset Owner": self.title_entry.get(),
                "Owner Department of Asset": self.owner_department_entry.get(),
                "Used by Process": self.process_entry.get(),
                "Confidentiality": self.confidentiality_combobox.get(),
                "Integrity": self.integrity_combobox.get(),
                "Availability": self.availability_combobox.get(),
                "Asset Value": self.value_combobox.get(),
                "Confidentiality Classification of Asset": self.conf_class_combobox.get(),
                "RTO": self.rto_entry.get(),
                "RPO": self.rpo_entry.get()
            }

            # Check if the edited quantity is a non-negative integer
            edited_quantity = edited_data["Quantity"]
            if edited_quantity.strip() != "":
                try:
                    edited_quantity = int(edited_quantity)
                    if edited_quantity < 0:
                        raise ValueError("Quantity must be a non-negative integer.")
                except ValueError as e:
                    # If the quantity is not a valid integer, show an error message
                    messagebox.showerror("Invalid Quantity", f"Invalid quantity: {e}")
                    return

            # Check if the entered ID is empty
            edited_id = edited_data["Product ID"]
            if not edited_id:
                # Assign the smallest unique ID to the item
                edited_id = str(self.get_smallest_unique_id())
            else:
                try:
                    edited_id = int(edited_id)
                except ValueError:
                    messagebox.showerror("Error", "Please enter a valid numeric ID.")
                    return

                if edited_id in self.inventory_df["Product ID"].values:
                    messagebox.showerror("Error", "Entered ID already exists. Please enter a unique ID.")
                    return

            # Check if a serial number was entered
            edited_serial = edited_data["Serial Number"]
            if edited_serial:
                if edited_serial in self.inventory_df["Serial Number"].values:
                    messagebox.showerror("Error",
                                         f"Serial Number '{edited_serial}' already exists! Please enter a different serial number.")
                    return

            # Update the inventory DataFrame with the edited data
            for column, edited_value in edited_data.items():
                if edited_value.strip() == "":
                    # If the edited value is empty, use the previous value from the DataFrame
                    self.inventory_df.at[index, column] = previous_data[column]
                else:
                    # Otherwise, use the edited value
                    self.inventory_df.at[index, column] = edited_value

            # Update the "Review Date" field with the current date
            self.inventory_df.at[index, "Review Date"] = datetime.datetime.now()

            # Update the treeview with the edited data
            self.update_table()

            # Show a message box to indicate the successful edit
            messagebox.showinfo("Edit Successful", "Item has been edited successfully.")

            # Clear the entry fields after editing
            self.clear_entry_fields()

            # Save the updated DataFrame to the Excel file
            self.save_to_excel()

        except Exception as e:
            # Handle any other unexpected errors
            messagebox.showerror("Error", f"An error occurred: {str(e)}")

    def search_items(self, event):
        # Get the search query from the entry field
        query = self.search_entry.get().lower()

        # Filter the inventory DataFrame based on the search query
        filtered_df = self.inventory_df[
            self.inventory_df.apply(lambda row: any(query in str(val).lower() for val in row), axis=1)
        ]

        # Update the treeview with the filtered results
        self.update_table(filtered_df)

        # Update the filtered_inventory_df only
        self.filtered_inventory_df = filtered_df.copy()

    def update_subcategories(self, event):
        # Get the selected main category from the maincategory_combobox
        selected_main_category = self.maincategory_combobox.get()

        # Clear the subcategory_combobox values
        self.subcategory_combobox.set("")  # Clear the current selection
        self.subcategory_combobox["values"] = []  # Clear the available options

        # Check if the selected main category has associated subcategories
        if selected_main_category:
            # Read the subcategories from the "rows.xlsx" file for the selected main category
            self.subcategory = self.read_column("rows.xlsx", "Sub Category of Asset", selected_main_category)

            # Filter out any NaN values from the subcategory list
            self.subcategory = [subcat for subcat in self.subcategory if pd.notna(subcat)]

            # Update the subcategory_combobox values with the new subcategories
            if self.subcategory:
                self.subcategory_combobox["values"] = self.subcategory
            else:
                self.subcategory_combobox["values"] = [""]
        else:
            self.subcategory_combobox["values"] = [""]

    def sort_column(self, col, reverse):
        """Sorts the Treeview by the specified column."""
        data = [(self.tree.set(child, col), child) for child in self.tree.get_children("")]
        data.sort(reverse=reverse)

        for index, item in enumerate(data):
            self.tree.move(item[1], "", index)

        # Change the sorting function for the next click
        self.tree.heading(col, command=lambda c=col: self.sort_column(c, not reverse))

    def remove_by_product_id(self):
        # Get the entered Product ID for removal
        product_id_to_remove = self.remove_by_product_id_entry.get()

        # Find the row in the DataFrame with the given Product ID
        try:
            product_id_to_remove = int(product_id_to_remove)
            row_to_remove = self.inventory_df[self.inventory_df["Product ID"] == product_id_to_remove]

            if row_to_remove.empty:
                messagebox.showerror("Error", f"Item with Product ID {product_id_to_remove} not found.")
                return

            # Remove the row from the DataFrame
            self.inventory_df = self.inventory_df.drop(row_to_remove.index)
            self.save_to_excel()
            self.update_table()  # Update the table to reflect the changes
            messagebox.showinfo("Success", f"Item with Product ID {product_id_to_remove} has been removed.")
        except ValueError:
            messagebox.showerror("Error", "Please enter a valid Product ID.")

    def get_smallest_unique_id(self):
        # Get the maximum existing product ID
        max_existing_id = self.inventory_df["Product ID"].max()

        for i in range(1, max_existing_id + 2):  # Loop from 1 to max_existing_id + 1
            if i not in self.inventory_df["Product ID"].values:
                return i
